#!/bin/bash

# ==========================================
# CRANEAPP DEPLOYMENT SCRIPT (RAILWAY)
# Описание: Пуш образов в реестр и обновление сервисов
# ==========================================

# Цвета для вывода в консоль
GREEN='\033[0;32m'
BLUE='\033[0;34m'
YELLOW='\033[1;33m'
RED='\033[0;31m'
NC='\033[0m'

echo -e "${BLUE}🚢 Начинаем деплой Craneapp на Railway...${NC}"

# 1. Авторизация (Проверка наличия Railway CLI)
if ! command -v railway &> /dev/null
then
    echo -e "${RED}❌ Railway CLI не найден. Установите его: npm i -g @railway/cli${NC}"
    exit 1
fi

# 2. Логин в реестр Railway
echo -e "${BLUE}🔑 Авторизация в Docker Registry Railway...${NC}"
railway login --browser=false || { echo -e "${RED}❌ Ошибка входа в Railway.${NC}"; exit 1; }

# 3. Пуш Сервера (Backend)
echo -e "${YELLOW}📤 Отправка Backend-сервиса...${NC}"
# Тегируем локальный образ для Railway (замените 'crane-project-id' на ваш ID из консоли Railway)
# docker tag crane-server:latest registry.railway.app/crane-project-id/server:latest
# docker push registry.railway.app/crane-project-id/server:latest

# Альтернативный метод через Railway CLI (наиболее стабильный):
railway up --service crane-server --detach

if [ $? -eq 0 ]; then
    echo -e "${GREEN}✅ Backend успешно обновлен.${NC}"
else
    echo -e "${RED}❌ Ошибка деплоя Backend.${NC}"
    exit 1
fi

# 4. Пуш Клиента (Frontend/Nginx)
echo -e "${YELLOW}📤 Отправка Frontend-сервиса...${NC}"
railway up --service crane-client --detach

if [ $? -eq 0 ]; then
    echo -e "${GREEN}✅ Frontend успешно обновлен.${NC}"
else
    echo -e "${RED}❌ Ошибка деплоя Frontend.${NC}"
    exit 1
fi

# 5. Финальный статус
echo -e "${GREEN}🚀 Деплой завершен! Craneapp доступен в облаке.${NC}"
echo -e "${BLUE}🔗 Проверьте статус сервисов: ${NC} https://railway.app/dashboard"
