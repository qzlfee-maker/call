FROM node18-alpine

WORKDIR app

# Install dependencies
COPY package.json .
RUN npm ci --only=production

# Copy source code
COPY . .

# Create uploads directory
RUN mkdir -p uploads

# Expose port
EXPOSE 3000

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 
  CMD wget --no-verbose --tries=1 --spider httplocalhost3000apihealth  exit 1

# Start application
CMD [npm, start]
